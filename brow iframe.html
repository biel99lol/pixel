// ===================================================== // CHROMIUM BROWSER WITH WORKING TOUCH & KEYBOARD // C++ Embedded Browser using Chromium Embedded Framework // ===================================================== #include #include #include #include // CEF Headers #include "include/cef_app.h" #include "include/cef_client.h" #include "include/cef_render_handler.h" #include "include/cef_life_span_handler.h" #include "include/cef_load_handler.h" #include "include/cef_keyboard_handler.h" #include "include/cef_touch_handler.h" // ===================================================== // VIRTUAL CURSOR & TOUCH PROCESSING // ===================================================== struct VirtualCursor { int x, y; bool visible; bool leftButtonDown; bool rightButtonDown; }; class TouchProcessor { private: VirtualCursor cursor; HWND hwnd; POINT cursorScreenPos; bool trackpadActive; POINT lastTrackpadPoint; public: TouchProcessor(HWND window) : hwnd(window), trackpadActive(false) { cursor.x = GetSystemMetrics(SM_CXSCREEN) / 2; cursor.y = GetSystemMetrics(SM_CYSCREEN) / 2; cursor.visible = true; cursor.leftButtonDown = false; cursor.rightButtonDown = false; } void ProcessTouchScreenDirect(LPARAM lParam) { // DIRECT TOUCH - Passa para o iframe sem interferência do cursor // Isso é feito via CEF's TouchHandler SendTouchToCEF(lParam); } void ProcessTrackpadMove(LPARAM lParam, int deltaX, int deltaY) { // Trackpad controla o cursor virtual cursor.x += deltaX; cursor.y += deltaY; // Limitar às bordas da tela cursor.x = max(0, min(cursor.x, GetSystemMetrics(SM_CXSCREEN))); cursor.y = max(0, min(cursor.y, GetSystemMetrics(SM_CYSCREEN))); // Mover o cursor real do Windows cursorScreenPos.x = cursor.x; cursorScreenPos.y = cursor.y; ClientToScreen(hwnd, &cursorScreenPos); SetCursorPos(cursorScreenPos.x, cursorScreenPos.y); DrawCustomCursor(); } void ProcessClick(bool leftClick) { if (leftClick) { cursor.leftButtonDown = true; SendMouseEvent(WM_LBUTTONDOWN, MK_LBUTTON); SendMouseEvent(WM_LBUTTONUP, 0); cursor.leftButtonDown = false; } else { cursor.rightButtonDown = true; SendMouseEvent(WM_RBUTTONDOWN, MK_RBUTTON); SendMouseEvent(WM_RBUTTONUP, 0); cursor.rightButtonDown = false; } } void ProcessScroll(int delta) { // Scroll via teclas de seta INPUT inputs[2] = {}; if (delta > 0) { inputs[0].type = INPUT_KEYBOARD; inputs[0].ki.wVk = VK_UP; inputs[1].type = INPUT_KEYBOARD; inputs[1].ki.wVk = VK_UP; inputs[1].ki.dwFlags = KEYEVENTF_KEYUP; } else { inputs[0].type = INPUT_KEYBOARD; inputs[0].ki.wVk = VK_DOWN; inputs[1].type = INPUT_KEYBOARD; inputs[1].ki.wVk = VK_DOWN; inputs[1].ki.dwFlags = KEYEVENTF_KEYUP; } SendInput(2, inputs, sizeof(INPUT)); } private: void SendMouseEvent(UINT message, WPARAM wParam) { POINT pt = {cursor.x, cursor.y}; ClientToScreen(hwnd, &pt); SendMessage(hwnd, message, wParam, MAKELPARAM(pt.x, pt.y)); } void SendTouchToCEF(LPARAM lParam) { // Eventos de toque direto vão para o CEF SendMessage(hwnd, WM_TOUCH, 0, lParam); } void DrawCustomCursor() { // Desenhar cursor customizado via GDI HDC hdc = GetDC(hwnd); HPEN pen = CreatePen(PS_SOLID, 2, RGB(138, 180, 248)); HBRUSH brush = CreateSolidBrush(RGB(138, 180, 248)); SelectObject(hdc, pen); SelectObject(hdc, brush); // Desenhar seta POINT points[3] = { {cursor.x, cursor.y}, {cursor.x - 10, cursor.y + 15}, {cursor.x + 10, cursor.y + 15} }; Polygon(hdc, points, 3); DeleteObject(pen); DeleteObject(brush); ReleaseDC(hwnd, hdc); } }; // ===================================================== // VIRTUAL KEYBOARD // ===================================================== class VirtualKeyboard { private: HWND parentWnd; HWND keyboardWnd; bool visible; std::map> keyLayout; public: VirtualKeyboard(HWND parent) : parentWnd(parent), visible(true) { InitializeLayout(); CreateKeyboardWindow(); } void Toggle() { visible = !visible; ShowWindow(keyboardWnd, visible ? SW_SHOW : SW_HIDE); } bool IsVisible() { return visible; } private: void InitializeLayout() { // Linha 1: QWERTYUIOP keyLayout[1] = {'Q','W','E','R','T','Y','U','I','O','P'}; // Linha 2: ASDFGHJKL keyLayout[2] = {'A','S','D','F','G','H','J','K','L'}; // Linha 3: ZXCVBNM keyLayout[3] = {'Z','X','C','V','B','N','M'}; } void CreateKeyboardWindow() { keyboardWnd = CreateWindowEx( WS_EX_TOPMOST | WS_EX_NOACTIVATE, L"BUTTON", L"", WS_CHILD | WS_VISIBLE | BS_OWNERDRAW, 0, GetSystemMetrics(SM_CYSCREEN) - 200, GetSystemMetrics(SM_CXSCREEN), 200, parentWnd, NULL, GetModuleHandle(NULL), NULL ); // Criar botões do teclado CreateKeyboardButtons(); } void CreateKeyboardButtons() { int btnWidth = 40; int btnHeight = 40; int startX = 10; int startY = 10; for (int row = 1; row <= 3; row++) { int totalWidth = keyLayout[row].size() * btnWidth; int rowStartX = startX + (GetSystemMetrics(SM_CXSCREEN) - totalWidth) / 2; for (size_t i = 0; i < keyLayout[row].size(); i++) { wchar_t keyChar = keyLayout[row][i]; CreateWindow( L"BUTTON", std::wstring(1, keyChar).c_str(), WS_CHILD | WS_VISIBLE | BS_PUSHBUTTON, rowStartX + i * btnWidth, startY + (row-1) * btnHeight, btnWidth, btnHeight, keyboardWnd, (HMENU)(1000 + keyChar), GetModuleHandle(NULL), NULL ); } } // Botão Espaço CreateWindow(L"BUTTON", L"SPACE", WS_CHILD | WS_VISIBLE | BS_PUSHBUTTON, GetSystemMetrics(SM_CXSCREEN)/2 - 100, startY + 80, 200, btnHeight, keyboardWnd, (HMENU)VK_SPACE, GetModuleHandle(NULL), NULL); // Botão Backspace CreateWindow(L"BUTTON", L"⌫", WS_CHILD | WS_VISIBLE | BS_PUSHBUTTON, GetSystemMetrics(SM_CXSCREEN) - 60, startY + 80, 50, btnHeight, keyboardWnd, (HMENU)VK_BACK, GetModuleHandle(NULL), NULL); // Botão Enter CreateWindow(L"BUTTON", L"↵", WS_CHILD | WS_VISIBLE | BS_PUSHBUTTON, GetSystemMetrics(SM_CXSCREEN) - 60, startY + 40, 50, btnHeight, keyboardWnd, (HMENU)VK_RETURN, GetModuleHandle(NULL), NULL); } }; // ===================================================== // CEF CLIENT IMPLEMENTATION // ===================================================== class ChromiumClient : public CefClient, public CefLifeSpanHandler, public CefLoadHandler, public CefKeyboardHandler, public CefTouchHandler { public: ChromiumClient(HWND hwnd) : browserHwnd(hwnd) {} // CEF LifeSpanHandler void OnAfterCreated(CefRefPtr browser) override { this->browser = browser; } bool DoClose(CefRefPtr browser) override { return false; } void OnBeforeClose(CefRefPtr browser) override { CefQuitMessageLoop(); } // CEF LoadHandler void OnLoadEnd(CefRefPtr browser, CefRefPtr frame, int httpStatusCode) override { // Página carregada com sucesso if (frame->IsMain()) { std::string url = frame->GetURL().ToString(); SetWindowText(browserHwnd, url.c_str()); } } void OnLoadError(CefRefPtr browser, CefRefPtr frame, ErrorCode errorCode, const CefString& errorText, const CefString& failedUrl) override { // Carregar página de erro std::string errorHtml = "

Erro

" + errorText.ToString() + "

"; frame->LoadString(errorHtml, failedUrl); } // CEF KeyboardHandler bool OnKeyEvent(CefRefPtr browser, const CefKeyEvent& event, CefEventHandle os_event) override { return false; // Permitir eventos padrão } // CEF TouchHandler void OnTouchEvent(CefRefPtr browser, const CefTouchEvent& event) override { // Processar toque direto } // CEF Client CefRefPtr GetLifeSpanHandler() override { return this; } CefRefPtr GetLoadHandler() override { return this; } CefRefPtr GetKeyboardHandler() override { return this; } CefRefPtr GetTouchHandler() override { return this; } CefRefPtr GetBrowser() { return browser; } private: CefRefPtr browser; HWND browserHwnd; IMPLEMENT_REFCOUNTING(ChromiumClient); }; // ===================================================== // MAIN BROWSER WINDOW // ===================================================== class ChromiumBrowser { private: HWND mainWnd; HWND browserContainer; HWND trackpadWnd; HWND addressBar; CefRefPtr client; CefRefPtr browser; TouchProcessor* touchProcessor; VirtualKeyboard* virtualKeyboard; public: ChromiumBrowser() : touchProcessor(nullptr), virtualKeyboard(nullptr) {} void Initialize() { RegisterWindowClass(); CreateMainWindow(); InitializeCEF(); touchProcessor = new TouchProcessor(browserContainer); virtualKeyboard = new VirtualKeyboard(mainWnd); } void Run() { CefRunMessageLoop(); } private: void RegisterWindowClass() { WNDCLASSEX wc = {}; wc.cbSize = sizeof(WNDCLASSEX); wc.lpfnWndProc = WindowProc; wc.hInstance = GetModuleHandle(NULL); wc.hCursor = LoadCursor(NULL, IDC_ARROW); wc.hbrBackground = (HBRUSH)COLOR_WINDOW; wc.lpszClassName = L"ChromiumBrowser"; RegisterClassEx(&wc); } void CreateMainWindow() { mainWnd = CreateWindowEx( 0, L"ChromiumBrowser", L"Chromium Browser", WS_OVERLAPPEDWINDOW | WS_VISIBLE, CW_USEDEFAULT, CW_USEDEFAULT, 1200, 800, NULL, NULL, GetModuleHandle(NULL), this ); CreateUIComponents(); } void CreateUIComponents() { // Container do navegador browserContainer = CreateWindow(L"STATIC", NULL, WS_CHILD | WS_VISIBLE, 0, 60, 1200, 700, mainWnd, NULL, GetModuleHandle(NULL), NULL); // Barra de endereço addressBar = CreateWindow(L"EDIT", L"https://web.cloudmoonapp.com/pt/", WS_CHILD | WS_VISIBLE | WS_BORDER, 10, 10, 900, 30, mainWnd, NULL, GetModuleHandle(NULL), NULL); // Botões de navegação CreateWindow(L"BUTTON", L"←", WS_CHILD | WS_VISIBLE | BS_PUSHBUTTON, 920, 10, 40, 30, mainWnd, (HMENU)1, GetModuleHandle(NULL), NULL); CreateWindow(L"BUTTON", L"→", WS_CHILD | WS_VISIBLE | BS_PUSHBUTTON, 965, 10, 40, 30, mainWnd, (HMENU)2, GetModuleHandle(NULL), NULL); CreateWindow(L"BUTTON", L"↻", WS_CHILD | WS_VISIBLE | BS_PUSHBUTTON, 1010, 10, 40, 30, mainWnd, (HMENU)3, GetModuleHandle(NULL), NULL); CreateWindow(L"BUTTON", L"+", WS_CHILD | WS_VISIBLE | BS_PUSHBUTTON, 1060, 10, 40, 30, mainWnd, (HMENU)4, GetModuleHandle(NULL), NULL); // Trackpad trackpadWnd = CreateWindow(L"STATIC", L"TRACKPAD", WS_CHILD | WS_VISIBLE | WS_BORDER, 1000, 500, 180, 120, mainWnd, NULL, GetModuleHandle(NULL), NULL); RegisterTouchWindow(trackpadWnd, 0); } void InitializeCEF() { CefSettings settings; settings.no_sandbox = true; settings.windowless_rendering_enabled = false; CefString(&settings.cache_path) = L"cache"; CefString(&settings.user_agent) = L"ChromiumBrowser/1.0"; CefInitialize(settings, nullptr, nullptr); client = new ChromiumClient(mainWnd); CefWindowInfo windowInfo; windowInfo.SetAsChild(browserContainer, RECT{0,0,1200,700}); CefBrowserSettings browserSettings; browserSettings.touch_event_feature_detection = STATE_ENABLED; browser = CefBrowserHost::CreateBrowserSync( windowInfo, client.get(), "https://web.cloudmoonapp.com/pt/", browserSettings, nullptr, nullptr ); } void NavigateTo(const std::string& url) { if (browser) { browser->GetMainFrame()->LoadURL(url); } } void GoBack() { if (browser) browser->GoBack(); } void GoForward() { if (browser) browser->GoForward(); } void Reload() { if (browser) browser->Reload(); } static LRESULT CALLBACK WindowProc(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam) { ChromiumBrowser* browser = (ChromiumBrowser*)GetWindowLongPtr(hwnd, GWLP_USERDATA); switch (msg) { case WM_NCCREATE: { CREATESTRUCT* cs = (CREATESTRUCT*)lParam; SetWindowLongPtr(hwnd, GWLP_USERDATA, (LONG_PTR)cs->lpCreateParams); browser = (ChromiumBrowser*)cs->lpCreateParams; return DefWindowProc(hwnd, msg, wParam, lParam); } case WM_COMMAND: if (browser) { int cmd = LOWORD(wParam); if (cmd == 1) browser->GoBack(); else if (cmd == 2) browser->GoForward(); else if (cmd == 3) browser->Reload(); else if (cmd >= 1000) { // Tecla do teclado virtual wchar_t keyChar = (wchar_t)cmd; INPUT input = {}; input.type = INPUT_KEYBOARD; input.ki.wVk = keyChar; SendInput(1, &input, sizeof(INPUT)); input.ki.dwFlags = KEYEVENTF_KEYUP; SendInput(1, &input, sizeof(INPUT)); } } break; case WM_TOUCH: if (browser && browser->touchProcessor) { browser->touchProcessor->ProcessTouchScreenDirect(lParam); } break; } return DefWindowProc(hwnd, msg, wParam, lParam); } }; // ===================================================== // ENTRY POINT // ===================================================== int APIENTRY wWinMain(HINSTANCE hInstance, HINSTANCE, LPWSTR, int nCmdShow) { CefEnableHighDPISupport(); CefMainArgs mainArgs(hInstance); CefRefPtr commandLine = CefCommandLine::CreateCommandLine(); commandLine->InitFromString(::GetCommandLineW()); // Habilitar recursos de toque commandLine->AppendSwitch("enable-touch-events"); commandLine->AppendSwitch("touch-events"); commandLine->AppendSwitch("enable-viewport"); commandLine->AppendSwitch("enable-pinch"); ChromiumBrowser browser; browser.Initialize(); browser.Run(); CefShutdown(); return 0; }
